# ─────────────────────────────────────────────
# Papa Bot v2 — Dockerfile
# Python 3.11 + Node.js + npm (as per runtime.txt & apt-packages)
# ─────────────────────────────────────────────

FROM python:3.11-slim

# System dependencies: nodejs & npm (as specified in apt-packages)
RUN apt-get update && apt-get install -y --no-install-recommends \
        nodejs \
        npm \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

# Working directory
WORKDIR /app

# Copy requirements first (layer caching)
COPY requirements.txt .

# Install Python dependencies
RUN pip install --no-cache-dir -r requirements.txt

# Copy bot source files
COPY main.py .
COPY file_manager.py .

# Port exposed by the Flask file manager (default 8080)
EXPOSE 8080

# Environment variables — set these at runtime via docker run -e or .env file
# Required:
#   BOT_TOKEN       — Telegram bot token
#   OWNER_ID        — Telegram owner user ID
#   MONGODB_URI     — MongoDB connection string
# Optional:
#   OWNER_USERNAME  — Telegram owner username (default: owner)
#   DATABASE_NAME   — MongoDB database name (default: god_madara_hosting)
#   BASE_URL        — Public base URL (default: http://localhost:8080)
#   PORT            — Flask port (default: 8080)

CMD ["python", "main.py"]
